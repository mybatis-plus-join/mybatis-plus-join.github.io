class MpJoinTest {
    @Resource
    private UserMapper userMapper;

    @Test
    void joinTest() {
        List<UserDTO> list = userMapper.selectJoinList(UserDTO.class, new MPJQueryWrapper<UserDO>()
                .selectAll(UserDO.class)
                .select("addr.tel")
                .leftJoin("user_address addr on addr.user_id = t.id")
                .eq("t.id", 1));
    }
}
